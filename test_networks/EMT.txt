AKT* = SMAD or SOS/GRB2 or not GSK3beta or beta_catenin_nuc
AXIN2* = AXIN2 or beta_catenin_nuc
beta_catenin_membrane* = E-cadherin and not beta_catenin_nuc
beta_catenin_nuc* = not Dest_compl and not beta_catenin_membrane and (SMAD or GLI or not E-cadherin) 
Dest_compl* = (GSK3beta and AXIN2 and beta_catenin_nuc) or (GSK3beta and Dest_compl)
pseudo* = not NOTCH and not SMAD
E-cadherin* = beta_catenin_membrane and (not SNAI1 or pseudo or not ZEB1 or not ZEB2 or not TWIST1 or not SNAI2)
GLI* = beta_catenin_nuc or SMAD or GLI
GSK3beta* = not GLI and not AKT
MEK* = SOS/GRB2 or not GSK3beta or beta_catenin_nuc or not (not MEK or not SNAI1)
NOTCH* = SOS/GRB2 or not GSK3beta or beta_catenin_nuc or SMAD
SMAD* =  (MEK or TGFbetaR) and (ZEB1 or not ZEB2)
SNAI1* = GLI or SMAD or beta_catenin_nuc or TGFbetaR or NOTCH or MEK or AKT or not GSK3beta
SNAI2* = MEK or beta_catenin_nuc or SNAI2 or TWIST1
SOS/GRB2* = (beta_catenin_nuc or TGFbetaR) and not MEK
miR200*  = not SNAI1 and not ZEB1 and not ZEB2
TGFbetaR* = SNAI1 or TWIST1 or GLI
TWIST1* = AKT or beta_catenin_nuc or SNAI1
ZEB1* = SNAI1 and not miR200
ZEB2*  = SNAI1 and not miR200
